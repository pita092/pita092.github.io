#include <stdio.h>
#define COLS 10
#define ROWS 10
#define CHAR '#'

int main() {
  int a[ROWS][COLS];



  for (int i = 0; i < ROWS; i++) {
    for (int j = 0; j < COLS; j++) {
      a[i][j] = CHAR;
    }
  }

  for (int i = 0; i < ROWS; i++) {
    for (int j = 0; j < COLS; j++) {
      printf(format: "%c", a[i][j]);
    }
    printf(format: "\n");
  }

  return 0;
}
